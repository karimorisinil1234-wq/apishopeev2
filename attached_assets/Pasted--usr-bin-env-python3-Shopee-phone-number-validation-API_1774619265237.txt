def pick_profile():
    """
    Versi ini DIPAKSA hanya pakai desktop UA:
    - 70% desktop Chrome (Windows/Mac)
    - 30% desktop Firefox/Safari
    """
    desktop_families = ["desktop_chrome", "firefox", "safari"]
    desktop_weights = [70, 20, 10]  # Chrome domian, sisanya desktop lain

    family = random.choices(
        population=desktop_families,
        weights=desktop_weights,
        k=1
    )[0]

    return random.choice(UA_GROUPS[family])

def build_headers(profile: dict):
    platform = profile["platform"]

    if platform == "Windows":
        sec_platform = '"Windows"'
    elif platform == "macOS":
        sec_platform = '"macOS"'
    elif platform == "Linux":
        sec_platform = '"Linux"'
    else:
        sec_platform = '"Windows"'

    headers = {
        "Accept-Language": "id-ID,id;q=0.9,en-US;q=0.8,en;q=0.7",
        "Accept": "text/html,application/xhtml+xml,application/xml;q=0.9,image/avif,image/webp,*/*;q=0.8",
        "Upgrade-Insecure-Requests": "1",
    }

    if "Chrome/" in profile["ua"]:
        headers["sec-ch-ua"] = '"Chromium";v="146", "Google Chrome";v="146", "Not.A/Brand";v="24"'
        headers["sec-ch-ua-mobile"] = "?0"
        headers["sec-ch-ua-platform"] = sec_platform
    elif "Firefox/" in profile["ua"]:
        # Firefox tidak pakai sec-ch-ua
        pass
    elif "Safari/" in profile["ua"]:
        # Safari desktop
        pass

    return headers


def make_context(browser):
    profile = pick_profile()
    context = browser.new_context(
        user_agent=profile["ua"],
        locale="id-ID",
        timezone_id="Asia/Jakarta",
        viewport=profile["viewport"],
        is_mobile=False,      # PAKSA DESKTOP
        has_touch=False,      # PAKSA DESKTOP
        extra_http_headers=build_headers(profile),
    )
    return context, profile


def page_looks_blocked(page) -> bool:
    try:
        current_url = page.url.lower()
        if "verify/traffic" in current_url:
            return True

        body_text = page.inner_text("body").lower()
        return any(k in body_text for k in BLOCKED_KEYWORDS)
    except Exception:
        return False

def dismiss_all_popups(page):
    popup_selectors = [
        # Bahasa
        "xpath=//button[contains(.,'Bahasa Indonesia')]",
        "xpath=//li[contains(.,'Bahasa Indonesia')]",
        "text=Bahasa Indonesia",

        # Cookie
        "xpath=//button[contains(.,'Terima')]",
        "xpath=//button[contains(.,'Accept')]",
        "xpath=//button[contains(.,'OK')]",
        "xpath=//button[@id='onetrust-accept-btn-handler']",

        # App banner / modal
        "xpath=//button[contains(.,'Nanti saja')]",
        "xpath=//button[contains(.,'Tidak, terima kasih')]",
        "xpath=//button[contains(.,'Lanjutkan di Browser')]",
        "xpath=//button[contains(.,'Continue in Browser')]",
        "xpath=//span[contains(.,'×')]//ancestor::button",
        "[aria-label='close']",
        "[aria-label='Close']",

        ".shopee-popup--close-btn",
        "[data-sqe='close']",
        ".modal-close",
    ]

    dismissed = 0
    for sel in popup_selectors:
        try:
            btn = page.query_selector(sel)
            if btn and btn.is_visible():
                btn.click()
                page.wait_for_timeout(random.randint(300, 700))
                dismissed += 1
        except Exception:
            continue

    return dismissed


def find_phone_input(page):
    try:
        page.wait_for_load_state("networkidle", timeout=10000)
    except Exception:
        pass

    page.wait_for_timeout(random.randint(800, 1500))

    dismiss_all_popups(page)
    page.wait_for_timeout(random.randint(400, 800))
    dismiss_all_popups(page)

    selectors = [
        "input[name='phoneOrEmail']",
        "input.X0Jdtz",
        "input[placeholder*='Telp']",
        "input[placeholder*='Email']",
        "input[placeholder*='No. Handphone']",
        "input[placeholder*='handphone']",
        "input[placeholder*='phone']",
        "input[type='text'][maxlength='128']",
        "input[type='text'][autocomplete='off']",
        "xpath=//input[@type='text' and not(@readonly)]",
    ]

    for attempt in range(3):
        for sel in selectors:
            try:
                el = page.wait_for_selector(sel, state="visible", timeout=5000)
                if el and el.is_visible():
                    return el
            except Exception:
                continue

        if attempt < 2:
            try:
                page.evaluate("window.scrollTo(0, 400)")
            except Exception:
                pass
            page.wait_for_timeout(random.randint(500, 1000))
            dismiss_all_popups(page)

    return None


def normalize_phone(raw: str) -> str:
    phone = raw.strip().replace(" ", "").replace("-", "").replace("(", "").replace(")", "")

    if phone.startswith("+62"):
        phone = phone[3:]
    elif phone.startswith("62") and len(phone) > 10:
        phone = phone[2:]
    elif phone.startswith("0"):
        phone = phone[1:]

    return phone

def check_phone(phone_number: str) -> dict:
    raw_input = phone_number.strip()
    local_number = normalize_phone(raw_input)
    display_number = "62" + local_number

    if not local_number.isdigit():
        return {"status": "error", "message": "Nomor harus berupa angka", "phone": display_number}
    if len(local_number) < 8 or len(local_number) > 13:
        return {"status": "error", "message": "Format nomor tidak valid", "phone": display_number}

    os.environ["PWDEBUG"] = "0"

    with sync_playwright() as p:
        browser = p.chromium.launch(headless=True, args=CHROME_ARGS)
        context, profile = make_context(browser)
        page = context.new_page()
        apply_stealth(page)

        try:
            page.goto(
                "https://shopee.co.id/buyer/reset?scenario=7",
                wait_until="domcontentloaded",
                timeout=30000
            )

            if page_looks_blocked(page):
                debug_save(page, "blocked_initial")
                return {
                    "status": "blocked",
                    "message": "Shopee mendeteksi trafik tidak wajar, coba lagi",
                    "phone": display_number,
                }

            phone_input = find_phone_input(page)
            if phone_input is None:
                debug_save(page, "input_not_found")
                return {
                    "status": "error",
                    "message": "Form input tidak ditemukan",
                    "phone": display_number,
                    "debug_url": page.url,
                }

            vp = profile["viewport"]
            page.mouse.move(
                random.randint(80, max(81, vp["width"] - 80)),
                random.randint(120, max(121, vp["height"] - 120))
            )
            page.wait_for_timeout(random.randint(250, 600))

            phone_input.click()
            page.wait_for_timeout(random.randint(150, 350))
            phone_input.click(click_count=3)
            page.wait_for_timeout(random.randint(100, 250))

            page.keyboard.press("Delete")
            page.wait_for_timeout(random.randint(80, 200))

            for ch in local_number:
                page.keyboard.type(ch, delay=random.randint(45, 120))

            page.wait_for_timeout(random.randint(400, 800))

            try:
                error_div = page.wait_for_selector(
                    "xpath=//div[starts-with(@id,'authentication-input-error_')]",
                    state="visible",
                    timeout=3000
                )
                error_msg = error_div.inner_text().strip()
                if "Nomor Telepon Tidak Sah" in error_msg or "Tidak Sah" in error_msg:
                    return {
                        "status": "invalid_format",
                        "message": error_msg,
                        "phone": display_number,
                        "registered": False,
                    }
            except Exception:
                pass

            try:
                next_btn = page.wait_for_selector(
                    "xpath=//button[contains(.,'Berikutnya') and not(@disabled)]",
                    state="visible",
                    timeout=15000
                )
            except PWTimeout:
                debug_save(page, "next_button_not_found")
                return {
                    "status": "error",
                    "message": "Tombol Berikutnya tidak muncul",
                    "phone": display_number,
                }

            page.wait_for_timeout(random.randint(300, 800))
            next_btn.click()
            page.wait_for_timeout(random.randint(1200, 2200))

            if page_looks_blocked(page):
                debug_save(page, "blocked_after_submit")
                return {
                    "status": "blocked",
                    "message": "Shopee mendeteksi trafik tidak wajar setelah submit",
                    "phone": display_number,
                }

            deadline = time.time() + 15
            while time.time() < deadline:
                page.wait_for_timeout(500)

                if page_looks_blocked(page):
                    debug_save(page, "blocked_polling")
                    return {
                        "status": "blocked",
                        "message": "Shopee mendeteksi trafik tidak wajar",
                        "phone": display_number,
                    }

                try:
                    warning_el = page.query_selector(".HyEuQL")
                    if warning_el:
                        warning_text = warning_el.inner_text().strip().lower()
                        if any(k in warning_text for k in NOT_REGISTERED_KEYWORDS):
                            return {
                                "status": "not_registered",
                                "message": "Nomor tidak terdaftar di Shopee",
                                "phone": display_number,
                                "registered": False,
                            }
                except Exception:
                    pass

                try:
                    body_text = page.inner_text("body").lower()

                    if any(k in body_text for k in NOT_REGISTERED_KEYWORDS):
                        return {
                            "status": "not_registered",
                            "message": "Nomor tidak terdaftar di Shopee",
                            "phone": display_number,
                            "registered": False,
                        }

                    otp_field = page.query_selector(
                        "input[name='otp'], "
                        "input[type='tel'][maxlength], "
                        "input[autocomplete='one-time-code']"
                    )
                    if otp_field:
                        return {
                            "status": "registered",
                            "message": "Nomor terdaftar di Shopee",
                            "phone": display_number,
                            "registered": True,
                        }

                    if any(k in body_text for k in REGISTERED_KEYWORDS):
                        return {
                            "status": "registered",
                            "message": "Nomor terdaftar di Shopee",
                            "phone": display_number,
                            "registered": True,
                        }
                except Exception:
                    pass

            debug_save(page, "status_unknown")
            return {
                "status": "unknown",
                "message": "Tidak bisa menentukan status, coba lagi",
                "phone": display_number,
                "registered": None,
            }

        except Exception as e:
            debug_save(page, "exception")
            return {"status": "error", "message": str(e), "phone": display_number}

        finally:
            context.close()
            browser.close()
